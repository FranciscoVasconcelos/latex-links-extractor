\makeatletter
\newcommand{\IfLabelExistsTF}[3]{\@ifundefined{r@#1}{#3}{#2}}
\makeatother

% Save original \hyperref and \ref
\let\orighyperref\hyperref
\let\origref\ref
\let\origpageref\pageref

% Redefine \hyperref
\renewcommand{\hyperref}[2][]{\IfLabelExistsTF{#1}{\orighyperref[#1]{#2}}{\href{\linkref{#1}}{#2}}}

% Create a new command for references
\newcommand{\wref}[1]{\IfLabelExistsTF{#1}{\ref{#1}}{\refweb{#1}}}

% Renew the command if you want
%\renewcommand{\ref}[1]{\IfLabelExistsTF{#1}{\origref{#1}}{\refweb{#1}}}

% Renew page referencing command
\renewcommand{\pageref}[1]{\IfLabelExistsTF{#1}{\origpageref{#1}}{\href{\linkref{#1}}{\pagewref{#1}}}}
